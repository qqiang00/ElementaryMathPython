def my_tan(p1, p2):
    dx = abs(p2[0] - p1[0])
    dy = abs(p2[1] - p1[1])
    if dx != 0:
        slope = dy/dx
    else:
        slope = float("inf")#"infinite"
    return slope

from math import sqrt
def my_sin(p1, p2):
    # p1, p2 are tuple: (x, y)
    # result = dy / distance(p1, p2)
    dy = abs(p2[1]-p1[1])
    dx = abs(p2[0]-p1[0])
    dis = sqrt(dx**2 + dy**2)
    result = dy / dis
    return result

from math import sqrt
def my_cos(p1, p2):
    # p1, p2 are tuple: (x, y)
    # result = dx / distance(p1, p2)
    dy = abs(p2[1]-p1[1])
    dx = abs(p2[0]-p1[0])
    dis = sqrt(dx**2 + dy**2)
    result = dx / dis
    return result
