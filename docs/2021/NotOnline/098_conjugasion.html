def get_person_key(person):
    if person in ["il", "elle", "on"]:
        return "il"
    if person in ["ils", "elles"]:
        return "ils"
    return person

def conjugaison_regular(verb):
    ending = get_ending(verb)
    for time in times:
        print("  {}  ".format(time))
        for person in persons:
            person_key = get_person_key(person)
            key = ending+"_"+time+"_"+person_key
            rule = rules[key]
            verb_conj = verb[:-1*rule[0]]   # remove suffix
            verb_conj = verb_conj + rule[1] # add extra
            print(person + " " + verb_conj) # connect person and verbe already changed.
        print()
