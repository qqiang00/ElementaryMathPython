Assignments 作业¶

  1. 现在有5个小朋友：Jason, Sophie, Tony, Yunzi, Celine, 他们排成一个纵队，那么一共有多少种排队方法？请分别用while循环for…in循环，和permutations方法来完成本题。 If now there is 5 chldren instead of 4 (Celine added), how many queues can be formed. Please use while loop, for…in loop, and permuations methods to answer this question seperatedly.

# Use while loop
classmates = ["Sophie", "Tony", "Jason", "Yunzi", "Celine"]
queues = []
n_classmates = len(classmates)
queue = []
index = 0
i, j, k, m, n = 0, 0, 0, 0, 0
while i < n_classmates:
    j = 0
    while j < n_classmates:
        if j != i:
            k = 0
            while k < n_classmates:
                if k != j and k != i:
                    m = 0
                    while m < n_classmates:
                        if m != k and m != j and m != i:
                            n = 0
                            while n < n_classmates:
                                if n != m and n != k and n != j and n != i:
                                    first = classmates[i]
                                    second = classmates[j]
                                    third = classmates[k]
                                    forth = classmates[m]
                                    fifth = classmates[n]
                                    queue = (first, second, third, forth, fifth)
                                    queues.append(queue)
                                    index += 1
                                n += 1
                        m += 1
                k += 1
        j += 1
    i += 1

for queue in queues:
    print(queue)

print("There are {} combinations.".format(index))

# Use method permutations
from itertools import permutations
classmates = ["Sophie", "Tony", "Jason", "Yunzi", "Celine"]
queues = []
candidates = list(permutations(classmates))
for candidate in candidates:
    queues.append(candidate)

for queue in queues:
    print(queue, end="\n")
print("{} queues".format(len(queues)))
