def three_equal_division(point1, point2):
    x1, y1 = point1
    x2, y2 = point2
    x_step = (x2 - x1)/3
    y_step = (y2 - y1)/3
    x3, y3 = x1 + x_step, y1 + y_step
    x4, y4 = x2 - x_step, y2 - y_step
    point3 = (x3, y3)
    point4 = (x4, y4)
    return [point3, point4]
